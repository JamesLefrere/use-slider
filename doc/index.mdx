import { Playground } from "docz";
import useSlider from "../src";
import "../src/example.scss";

# Button

<Playground>
  {() => {
    const [ref, slide] = useSlider();
    return (
      <>
        <div ref={ref}>
          <div>1</div>
          <div>2</div>
          <div>3</div>
          <div>4</div>
          <div>5</div>
        </div>
        <button onClick={() => slide.prev()}>Prev</button>
        <button onClick={() => slide.next()}>Next</button>
      </>
    );
  }}
</Playground>
