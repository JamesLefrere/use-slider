import { Playground } from "docz";
import useSlider from "../src";
import "../src/slider.scss";

# Multiple Slide With Loop

<Playground>
  {() => {
    const [ref, slide] = useSlider({ slidesPerView: 3, loop: true });
    const list = [1, 2, 3, 4];
    return (
      <div ref={ref}>
        {list.map(item => (
          <div key={item}>{item}</div>
        ))}
      </div>
    );
  }}
</Playground>
